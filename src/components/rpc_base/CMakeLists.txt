add_library(${PROJECT_NAME} STATIC)

  PRIVATE
    "${CMAKE_CURRENT_LIST_DIR}/src/rpc_base/rpc_base.cc"

  PUBLIC
    jsoncpp::jsoncpp
    Policy::policy_table
  PRIVATE
    components::interface
)

  PUBLIC "${CMAKE_CURRENT_LIST_DIR}/include"
)
